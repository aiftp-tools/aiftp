```ts
// packages/cli/src/prompt-framework/types.ts

export type PromptFieldType = 'text' | 'password' | 'number' | 'select' | 'confirm';

export interface PromptFieldChoice {
  title: string;
  value: string;
  description?: string;
}

export interface PromptField<T = unknown> {
  /** フィールド識別子（answers キー） */
  name: string;
  /** ユーザー向けラベル（prompt 質問文） */
  label: string;
  /** prompts ライブラリの type と一致 */
  type: PromptFieldType;
  /** A: 入力ヒント（説明文）— 質問前に stderr に表示 */
  hint?: string;
  /** A: 具体例（"21 (FTP), 990 (FTPS)" 等）— 質問前に表示 */
  example?: string;
  /** zod や独自 validator — true | エラーメッセージ */
  validate?: (value: T, answers: Record<string, unknown>) => true | string;
  /** raw 入力 → クリーンな T への変換（trim, control char reject 等） */
  sanitize?: (raw: unknown) => T;
  /** prompts ライブラリの initial 値 */
  initial?: T | ((answers: Record<string, unknown>) => T);
  /** select type 専用 */
  choices?: PromptFieldChoice[];
  /** number type 専用 */
  min?: number;
  max?: number;
}

export type PromptResult<T> =
  | { kind: 'value'; value: T }
  | { kind: 'back' }
  | { kind: 'cancel' };

export type FlowResult =
  | { kind: 'completed'; answers: Record<string, unknown> }
  | { kind: 'cancelled' };

export const BACK_KEYWORD = ':back';
```

```ts
// packages/cli/src/prompt-framework/prompt-flow.ts
import type { FlowResult, PromptField } from './types.ts';

export interface PromptFlowDeps {
  prompt: (q: unknown) => Promise<Record<string, unknown>>;
  stderr: (msg: string) => void;
}

export class PromptFlow {
  constructor(
    private readonly fields: ReadonlyArray<PromptField>,
    private readonly deps: PromptFlowDeps,
  ) {}

  async run(): Promise<FlowResult> {
    const answers: Record<string, unknown> = {};
    for (const field of this.fields) {
      const raw = await this.deps.prompt({
        name: field.name,
        type: field.type,
        message: field.label,
      });
      answers[field.name] = raw[field.name];
    }
    return { kind: 'completed', answers };
  }
}
```

```ts
// prompt-flow.ts を以下のように書き直す
async run(): Promise<FlowResult> {
  const answers: Record<string, unknown> = {};
  let cursor = 0;
  while (cursor < this.fields.length) {
    const field = this.fields[cursor];
    this.printHint(field);
    const raw = await this.deps.prompt({
      name: field.name,
      type: 'text',  // :back 受理のため number/select でも text 扱い、後で field.type でパース
      message: field.label,
    });
    const value = raw[field.name];

    if (value === null || value === undefined) return { kind: 'cancelled' };
    if (value === BACK_KEYWORD) {
      if (cursor === 0) {
        this.deps.stderr('  ↩ もう戻れません（1問目です）');
      } else {
        cursor--;
      }
      continue;
    }

    answers[field.name] = value;
    cursor++;
  }
  return { kind: 'completed', answers };
}
```

```ts
  const answers: Record<string, unknown> = {};
  let cursor = 0;
  while (cursor < this.fields.length) {
    const field = this.fields[cursor];
    this.printHint(field);
    const raw = await this.deps.prompt({
      name: field.name, type: 'text', message: field.label,
    });
    const value = raw[field.name];

    if (value === null || value === undefined) return { kind: 'cancelled' };
    if (value === BACK_KEYWORD) {
      if (cursor === 0) this.deps.stderr('  ↩ もう戻れません（1問目です）');
      else cursor--;
      continue;
    }

    // sanitize
    const sanitized = field.sanitize ? field.sanitize(value) : value;
    // validate
    if (field.validate) {
      const v = field.validate(sanitized, answers);
      if (v !== true) {
        this.deps.stderr(`  ✗ ${v}`);
        continue;  // 同じ field を再 prompt
      }
    }

    answers[field.name] = sanitized;
    cursor++;
  }
  return { kind: 'completed', answers };
}
```

```ts
// sftp-client.ts
import Sftp from 'ssh2-sftp-client';
import type { FtpClientOptions, FileInfo } from './ftp-client.ts';

export class SftpClient {
  private client: Sftp | null = null;
  constructor(private readonly options: FtpClientOptions) {}

  async connect(): Promise<void> {
    this.client = new Sftp();
    await this.client.connect({
      host: this.options.host,
      port: this.options.port ?? 22,
      username: this.options.user,
      password: this.options.password,
      // SSH 鍵は Task 23 で追加
    });
  }

  async disconnect(): Promise<void> {
    if (this.client) await this.client.end();
    this.client = null;
  }

  async list(remotePath: string): Promise<FileInfo[]> {
    if (!this.client) throw new Error('not connected');
    const items = await this.client.list(remotePath);
    return items.map((i) => ({
      name: i.name,
      type: i.type === 'd' ? 'directory' : 'file',
      size: i.size,
      modifiedAt: new Date(i.modifyTime),
    }));
  }
}
```

```ts
// sftp-client.ts
import { readFileSync, statSync } from 'node:fs';

async connect(): Promise<void> {
  this.client = new Sftp();
  const config: Record<string, unknown> = {
    host: this.options.host, port: this.options.port ?? 22, username: this.options.user,
  };
  if (this.options.sshKeyPath) {
    const mode = statSync(this.options.sshKeyPath).mode & 0o777;
    if (mode !== 0o600 && mode !== 0o400) {
      throw new Error(`SSH key permissions must be 600 or 400, got ${mode.toString(8)}`);
    }
    config.privateKey = readFileSync(this.options.sshKeyPath);
  } else if (this.options.password) {
    config.password = this.options.password;
  } else {
    throw new Error('Either password or sshKeyPath must be provided for SFTP');
  }
  await this.client.connect(config);
}
```

```ts
// deploy-client-factory.ts
import { FtpClient } from './ftp-client.ts';
import { SftpClient } from './sftp-client.ts';
import type { FtpClientOptions } from './ftp-client.ts';

export type DeployClient = FtpClient | SftpClient;

export function createDeployClient(options: FtpClientOptions & { protocol: 'ftp' | 'ftps' | 'sftp' }): DeployClient {
  switch (options.protocol) {
    case 'ftp':
    case 'ftps':
      return new FtpClient(options);
    case 'sftp':
      return new SftpClient(options);
    default:
      throw new Error(`unknown protocol: ${options.protocol}`);
  }
}
```

```markdown
- **PromptFlow framework** (`packages/cli/src/prompt-framework/`) — ステートマシン型 prompt runner。入力ヒント表示 (A) と `:back` 戻りナビ (B) を実装。三重防御 A/B/C が完成。v0.11 では `aiftp init` のみ移行、`aiftp auth` / `aiftp config edit` 等 30 箇所は v0.11.x 以降で順次移行。
- **WordPress 特化テンプレ機能** (`packages/core/src/templates/`) — 7 種（wordpress-swell / wordpress-lightning / wordpress-cocoon / wordpress-standard / static / laravel / php-simple）。`aiftp init --template wordpress-swell` で 30 秒 onboarding。
- **SFTP 対応** (`packages/core/src/sftp-client.ts`) — ssh2-sftp-client 経由で SSH-based deploy をサポート。VPS / 専用サーバーへの展開を可能に。
- **doctor SFTP 4 check** — `ssh-port-reachable` / `ssh-key-permissions` / `sftp-handshake` / `sftp-remote-root`
- **MCP `aiftp_init_template_list` tool** — テンプレ一覧の read-only 提供。
- **smoke test CI** — 3 OS (macOS/Ubuntu/Windows) × 2 Node (22/24) matrix、release/dispatch/schedule トリガー。
- **競合比較ドキュメント** — vs alxspiker / Computer Use / WordPress 公式 MCP。

### Changed

- `parseInitAnswers` は内部実装、PromptFlow 統合で `runInitFlow()` ヘルパー経由になる。
- `deploy.ts` / `rollback.ts` / `backup/` の FtpClient 直接 import は `deploy-client-factory.ts` 経由に変更。protocol が `sftp` の場合は SftpClient に切替。
```
